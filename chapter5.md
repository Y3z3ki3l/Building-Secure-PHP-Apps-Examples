##CSRF

###Token Generation

	class Form
	{

		static function generateCsrf() {

		  $token = mcrypt_create_iv(
		    16,
		    MCRYPT_DEV_URANDOM
		  );

		  Session::flash(
		    'csrfToken',
		    $token
		  );


		  return $token;
		}

	}


###Generate token for form

	Route::get('/signup', function(){

	  $data['token'] = Form::generateCsrf();


	  return View::render('signup.form', $data);

	});
